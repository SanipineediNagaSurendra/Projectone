pipeline {
    agent any

    environment {
        DOTNET_VERSION = '8.0' // update this to match your actual .NET version
    }

    stages {
        stage('Checkout Code') {
            steps {
                echo '🔄 Checking out the code...'
                git url: 'https://github.com/SanipineediNagaSurendra/Projectone.git'
            }
        }

        stage('Restore Dependencies') {
            steps {
                echo '🔧 Restoring .NET dependencies...'
                bat 'dotnet restore'
            }
        }

        stage('Install Playwright CLI') {
            steps {
                echo '🔧 Installing Playwright CLI locally...'
                bat '''
                    dotnet new tool-manifest --force
                    dotnet tool install Microsoft.Playwright.CLI
                '''
            }
        }

        stage('Build') {
            steps {
                echo '⚙️ Building the project...'
                bat 'dotnet build --configuration Release'
            }
        }

        stage('Install Playwright Browsers') {
            steps {
                echo '🌐 Installing Playwright browsers...'
                bat 'dotnet tool run playwright install'
            }
        }

        stage('Test') {
            steps {
                echo '🧪 Running tests...'
                bat 'dotnet test --configuration Release --logger "trx;LogFileName=testresults.trx"'
            }
        }
    }

    post {
        success {
            echo '✅ Build and tests completed successfully!'
        }
        failure {
            echo '❌ Pipeline failed. Please check the logs.'
        }
        always {
            echo '🏁 Pipeline run complete.'
            // Optional: if you install MSTest plugin, convert .trx to .xml and use:
            // junit '**/TestResults/*.xml'
        }
    }
}
